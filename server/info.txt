# Make deck of cards
def create_deck():
    deck = []
    for color in COLORS:
        for value in VALUES:
            deck.append(f"{color} {value}")
            if value != '0':
                deck.append(f"{color} {value}")
    for i in range(4):
        deck.append("Wild")
        deck.append("Draw Four")
    return deck
